-- ZOONOSEAPP MVP -- SCHEMA COMPLETO

CREATE TABLE sectors (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name         TEXT NOT NULL,
  municipality TEXT NOT NULL DEFAULT 'Uberlandia',
  created_at   TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE profiles (
  id         UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  name       TEXT NOT NULL,
  role       TEXT NOT NULL CHECK (role IN ('agent','coordinator','manager','admin')),
  sector_id  UUID REFERENCES sectors(id),
  active     BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE OR REPLACE FUNCTION handle_new_user() RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO profiles (id, name, role) VALUES
  (NEW.id, NEW.raw_user_meta_data->>'name', COALESCE(NEW.raw_user_meta_data->>'role', 'agent'));
RETURN NEW; END; $$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();

CREATE TABLE properties (
  id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  code       TEXT UNIQUE NOT NULL,
  address    TEXT NOT NULL,
  sector_id  UUID REFERENCES sectors(id),
  type       TEXT DEFAULT 'residencial'
               CHECK (type IN ('residencial','comercial','terreno','especial')),
  lat        DOUBLE PRECISION,
  lng        DOUBLE PRECISION,
  active     BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE visits (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  property_id UUID NOT NULL REFERENCES properties(id),
  agent_id    UUID NOT NULL REFERENCES profiles(id),
  status      TEXT NOT NULL CHECK (status IN (
                'visitado_sem_foco','visitado_com_achado','fechado',
                'recusado','nao_localizado','pendente_revisao')),
  notes       TEXT,
  visited_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  synced_at   TIMESTAMPTZ,
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE evidences (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  visit_id     UUID NOT NULL REFERENCES visits(id) ON DELETE CASCADE,
  storage_path TEXT NOT NULL,
  created_at   TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE activity_logs (
  id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id    UUID REFERENCES profiles(id),
  action     TEXT NOT NULL,
  entity     TEXT, entity_id UUID,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- RLS POLICIES
ALTER TABLE visits     ENABLE ROW LEVEL SECURITY;
ALTER TABLE profiles   ENABLE ROW LEVEL SECURITY;
ALTER TABLE properties ENABLE ROW LEVEL SECURITY;
ALTER TABLE evidences  ENABLE ROW LEVEL SECURITY;

CREATE POLICY "profiles_self" ON profiles FOR SELECT USING
  (id = auth.uid() OR EXISTS (SELECT 1 FROM profiles p
   WHERE p.id = auth.uid() AND p.role IN ('admin','manager')));

CREATE POLICY "properties_read" ON properties FOR SELECT USING (auth.role() = 'authenticated');

CREATE POLICY "visits_read" ON visits FOR SELECT USING (
  agent_id = auth.uid() OR EXISTS (SELECT 1 FROM profiles p
  WHERE p.id = auth.uid() AND p.role IN ('coordinator','manager','admin')));

CREATE POLICY "visits_insert" ON visits FOR INSERT WITH CHECK (agent_id = auth.uid());

-- Indices

import { MMKV } from 'react-native-mmkv'
export const storage = new MMKV()

export const offlineQueue = {
  getAll: (): PendingVisit[] => {
    const raw = storage.getString('offline_queue')
    return raw ? JSON.parse(raw) : []
  },
  add: (visit: PendingVisit) => {
    const q = offlineQueue.getAll(); q.push(visit)
    storage.set('offline_queue', JSON.stringify(q))
  },
  remove: (id: string) => {
    storage.set('offline_queue',
      JSON.stringify(offlineQueue.getAll().filter(v => v.localId !== id)))
  },
  clear: () => storage.delete('offline_queue'),
}

interface PendingVisit {
  localId: string; property_id: string; agent_id: string
  status: string; notes?: string; visited_at: string; photoUri?: string
}

import { useEffect, useCallback } from 'react'
import NetInfo from '@react-native-community/netinfo'
import { supabase } from '../lib/supabase'
import { offlineQueue } from '../lib/mmkv'

export function useOfflineSync() {
  const syncPending = useCallback(async () => {
    const queue = offlineQueue.getAll()
    if (queue.length === 0) return
    for (const item of queue) {
      try {
        const { error } = await supabase.from('visits').insert({
          ...item, synced_at: new Date().toISOString(),
        })
        if (!error) offlineQueue.remove(item.localId)
      } catch (e) { console.warn('Sync falhou', item.localId, e) }
    }
  }, [])

  useEffect(() => {
    const unsub = NetInfo.addEventListener(s => { if (s.isConnected) syncPending() })
    return unsub
  }, [syncPending])

  return { syncPending }
}
📄 app/src/hooks/useAuth.ts
import { useState, useEffect } from 'react'
import { supabase } from '../lib/supabase'
import { Profile } from '../types'

export function useAuth() {
  const [profile, setProfile] = useState<Profile | null>(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    const { data: { subscription } } = supabase.auth.onAuthStateChange(async (_e, session) => {
      if (session?.user) {
        const { data } = await supabase
          .from('profiles').select('*').eq('id', session.user.id).single()
        setProfile(data)
      } else { setProfile(null) }
      setLoading(false)
    })
    return () => subscription.unsubscribe()
  }, [])

  const signIn = async (email: string, password: string) => {
    const { error } = await supabase.auth.signInWithPassword({ email, password })
    return { error }
  }
  const signOut = async () => { await supabase.auth.signOut() }
  return { profile, loading, signIn, signOut }
}

import React, { useState, useEffect } from 'react'
import { View, Text, TextInput, FlatList, TouchableOpacity, StyleSheet, ActivityIndicator } from 'react-native'
import { useNavigation } from '@react-navigation/native'
import { supabase } from '../lib/supabase'
import { offlineQueue } from '../lib/mmkv'
import { Property } from '../types'

export default function HomeScreen() {
  const nav = useNavigation()
  const [query, setQuery] = useState('')
  const [properties, setProperties] = useState<Property[]>([])
  const [loading, setLoading] = useState(false)
  const pendingCount = offlineQueue.getAll().length

  const search = async (q: string) => {
    if (q.length < 2) { setProperties([]); return }
    setLoading(true)
    const { data } = await supabase
      .from('properties').select('*')
      .or(`code.ilike.%${q}%,address.ilike.%${q}%`)
      .eq('active', true).limit(20)
    setProperties(data ?? [])
    setLoading(false)
  }

  useEffect(() => { search(query) }, [query])

  return (
    <View style={s.container}>
      {pendingCount > 0 && (
        <TouchableOpacity style={s.syncBanner}
          onPress={() => nav.navigate('SyncStatus' as any)}>
          <Text style={s.syncBannerText}>⚡ {pendingCount} visita(s) pendente(s) de sync</Text>
        </TouchableOpacity>
      )}
      <TextInput style={s.search} placeholder="Buscar por codigo ou endereco..."
        placeholderTextColor="#64748b" value={query} onChangeText={setQuery} />
      {loading && <ActivityIndicator color="#3b82f6" style={{ marginTop: 16 }} />}
      <FlatList data={properties} keyExtractor={p => p.id}
        renderItem={({ item }) => (
          <TouchableOpacity style={s.item}
            onPress={() => nav.navigate('PropertyDetail' as any, { propertyId: item.id })}>
            <Text style={s.itemCode}>{item.code}</Text>
            <Text style={s.itemAddress} numberOfLines={1}>{item.address}</Text>
          </TouchableOpacity>
        )} />
    </View>
  )
}

import React, { useState } from 'react'
import { View, Text, TouchableOpacity, TextInput, ScrollView, Alert, ActivityIndicator } from 'react-native'
import { useRoute, useNavigation } from '@react-navigation/native'
import NetInfo from '@react-native-community/netinfo'
import * as ImagePicker from 'expo-image-picker'
import { supabase } from '../lib/supabase'
import { offlineQueue } from '../lib/mmkv'
import { useAuth } from '../hooks/useAuth'
import { VisitStatus, STATUS_CONFIG } from '../types'

export default function RegisterVisitScreen() {
  const route = useRoute() as any
  const nav = useNavigation()
  const { profile } = useAuth()
  const [selectedStatus, setSelectedStatus] = useState<VisitStatus | null>(null)
  const [notes, setNotes] = useState('')
  const [photoUri, setPhotoUri] = useState<string | null>(null)
  const [saving, setSaving] = useState(false)

  const pickPhoto = async () => {
    const result = await ImagePicker.launchCameraAsync({ quality: 0.6 })
    if (!result.canceled) setPhotoUri(result.assets[0].uri)
  }

  const handleSave = async () => {
    if (!selectedStatus) { Alert.alert('Selecione o status'); return }
    if (!profile) return
    setSaving(true)
    const state = await NetInfo.fetch()
    const visitData = {
      property_id: route.params.propertyId,
      agent_id: profile.id, status: selectedStatus,
      notes: notes || undefined,
      visited_at: new Date().toISOString(),
    }
    if (state.isConnected) {
      const { error } = await supabase.from('visits').insert({
        ...visitData, synced_at: new Date().toISOString()
      })
      if (error) { Alert.alert('Erro', error.message); setSaving(false); return }
    } else {
      offlineQueue.add({ ...visitData, localId: Date.now().toString(), photoUri: photoUri ?? undefined })
    }
    setSaving(false)
    nav.navigate('ConfirmVisit' as any, { visitData: { ...visitData, offline: !state.isConnected } })
  }

  return (
    <ScrollView style={{ flex: 1, backgroundColor: '#0a1628' }}
      contentContainerStyle={{ padding: 16, paddingBottom: 40 }}>
      <Text style={{ color: '#94a3b8' }}>
        Imovel: <Text style={{ color: '#60a5fa' }}>{route.params?.propertyCode}</Text>
      </Text>
      <Text style={{ color: '#94a3b8', marginTop: 20, marginBottom: 10, fontWeight: '700' }}>
        STATUS DA VISITA
      </Text>
      <View style={{ flexDirection: 'row', flexWrap: 'wrap', gap: 8, marginBottom: 20 }}>
        {(Object.entries(STATUS_CONFIG) as [VisitStatus, any][]).map(([key, cfg]) => (
          <TouchableOpacity key={key}
            style={{ borderWidth: 2, borderRadius: 10, paddingHorizontal: 14, paddingVertical: 10,
              minWidth: '47%', borderColor: cfg.color,
              backgroundColor: selectedStatus === key ? cfg.bg : 'transparent' }}
            onPress={() => setSelectedStatus(key)}>
            <Text style={{ color: cfg.color, fontWeight: '700', textAlign: 'center' }}>{cfg.label}</Text>
          </TouchableOpacity>
        ))}

async function getMetrics() {
  const supabase = createClient()
  const today = new Date().toISOString().split('T')[0]

import { NextRequest, NextResponse } from 'next/server'
import { createClient } from '@/lib/supabase'

export async function GET(req: NextRequest) {
  const supabase = createClient()
  const days = req.nextUrl.searchParams.get('days') ?? '30'
  const since = new Date()
  since.setDate(since.getDate() - parseInt(days))

  const { data, error } = await supabase
    .from('visits')
    .select('visited_at, status, notes, properties(code, address, type), profiles(name)')
    .gte('visited_at', since.toISOString())
    .order('visited_at', { ascending: false })

  if (error) return NextResponse.json({ error: error.message }, { status: 500 })

  const header = 'Data,Hora,Codigo,Endereco,Tipo,Status,Agente,Nota\n'
  const rows = (data ?? []).map((v: any) => {
    const d = new Date(v.visited_at)
    return [
      d.toLocaleDateString('pt-BR'),
      d.toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' }),
      v.properties?.code ?? '',
      `"${v.properties?.address ?? ''}"`,
      v.properties?.type ?? '', v.status,
      v.profiles?.name ?? '',
      `"${(v.notes ?? '').replace(/"/g, '""')}"`,
    ].join(',')
  }).join('\n')

  return new NextResponse(header + rows, {
    headers: {
      "Content-Type": 'text/csv; charset=utf-8',
      "Content-Disposition": `attachment; filename="visitas_${days}dias.csv"`,
    },
  })
}

// USO: node import-imoveis.js imoveis.csv
// Formato: codigo,endereco,setor_nome,tipo
// Requer: npm install @supabase/supabase-js csv-parse dotenv

require('dotenv').config({ path: '../app/.env' })
const { createClient } = require('@supabase/supabase-js')
const { parse } = require('csv-parse/sync')
const fs = require('fs')

const supabase = createClient(
  process.env.EXPO_PUBLIC_SUPABASE_URL,
  process.env.EXPO_PUBLIC_SUPABASE_ANON_KEY
)

async function main() {
  const file = process.argv[2]
  if (!file) { console.error('Informe o arquivo CSV'); process.exit(1) }

  const content = fs.readFileSync(file, 'utf8')
  const records = parse(content, { columns: true, skip_empty_lines: true })

  const { data: sectors } = await supabase.from('sectors').select('id, name')
  const sectorMap = Object.fromEntries(sectors.map(s => [s.name.toLowerCase(), s.id]))

  const properties = records
    .map(r => ({
      code:      r.codigo?.trim(),
      address:   r.endereco?.trim(),
      sector_id: sectorMap[r.setor_nome?.trim().toLowerCase()],
      type:      r.tipo?.trim() || 'residencial',
    }))
    .filter(p => p.code && p.address)

  console.log(`Importando ${properties.length} imoveis...`)

  const { error } = await supabase
    .from('properties')
    .upsert(properties, { onConflict: 'code' })

  if (error) { console.error('Erro:', error.message); process.exit(1) }
  console.log(`OK: ${properties.length} imoveis importados com sucesso.`)
}
